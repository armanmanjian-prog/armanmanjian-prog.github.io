Full Example

#include <Arduino_MKRIoTCarrier.h>

MKRIoTCarrier carrier;
int led = 5;

void setup(){
 Serial.begin(9600);
 carrier.begin();
 pinMode(led, OUTPUT);
}

void loop(){
 int v = analogRead(A0);

 if(v > 500){
  digitalWrite(led, HIGH);
 } else {
  digitalWrite(led, LOW);
 }

 Serial.println(v);
 delay(100);
}

Tips

  • Always use carrier.begin()
  • Use Serial.println() for debugging
  • Use functions to organize code
  • Avoid long delay()
